534 void ChargedDReconstruction::saveDpInfo(CDDecayList::iterator it, double ebeam, int numofchildren,
535  EvtRecDTag* recDTag)
536 {
537 
538  double mass = (*it).particle().mass();
539  int charge = (*it).particle().charge();
540  HepLorentzVector p4((*it).particle().momentum(), (*it).particle().energy());
541  recDTag->setp4(p4);
542 
543  p4.boost(-m_beta);
544  double mbc2_CMS = ebeam * ebeam - p4.v().mag2();
545  double mbc_CMS = mbc2_CMS > 0 ? sqrt(mbc2_CMS) : -10;
546  double deltaE_CMS = p4.t() - ebeam;
547 
548  if((*it).particle().userTag() == 1)
549  recDTag->settype(EvtRecDTag::Tight);
550  else
551  recDTag->settype(EvtRecDTag::Loose);
552  recDTag->setcharge(charge);
553  recDTag->setcharm(charge);
554  recDTag->setnumOfChildren(numofchildren);
555  recDTag->setmass(mass);
556  recDTag->setmBC(mbc_CMS);
557  recDTag->setbeamE(ebeam);
558  recDTag->setdeltaE(deltaE_CMS);
559 }
