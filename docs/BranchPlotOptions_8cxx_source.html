40  const bool BranchPlotOptions::Import(TString input)
41  {
43  Int_t nsemicol { input.CountChar(';') };
44  if(ImportWarning(input, "has to contain at least 3 semicolumns (at least 4 argument blocks)", nsemicol < 3)) return false;
45  if(ImportWarning(input, Form("contains %d branches, which is impossible to plot", nsemicol-2), nsemicol > 5)) return false;
46  Ssiz_t count{0};
48  while(input.Contains(";;")) input.Insert(input.Index(";;")+1, " ");
49  if(input.EndsWith(";")) input+=" ";
51  TString tok;
52  Ssiz_t start{0};
53  if(ImportWarning(input, "does not contain a tree name", !Tokenize(input, tok, start))) return false;
54  SetTreeName(tok);
55  ++count;
57  while(count < nsemicol-1 && Tokenize(input, tok, start)) {
58  if(!AddBranch(tok)) return false;
59  ++count;
60  }
62  if(ImportWarning(input, "does not contain a cut selection", !Tokenize(input, tok, start))) return false;
63  SetCutSelection(tok);
65  if(ImportWarning(input, "does not contain draw options", !Tokenize(input, tok, start))) return false;
66  SetDrawOptions(tok);
68  fIsOK = true;
69  return true;
70  }

96  void BranchPlotOptions::SetDrawOptions(TString input)
97  {
100  input.ToLower();
102  auto start{input.Index("log")};
103  if(start > 0) {
104  TString tok;
105  for(auto i : fSetLog) i = false; // reset bits
106  while(input.Tokenize(tok, start, "log")) {
107  for(int i=0; i<tok.Length(); ++i) {
108  UChar_t index = tok[i]-'x'; // unsiged to simplify next if statement
109  if(index>=3) break; // if not x, y, or z
110  fSetLog[index] = true;
111  }
112  }
113  TRegexp search{"log[xyz]*"};
114  while(input.Contains(search)) {
115  input.Remove(input.Index(search), input(search).Length());
116  }
117  }
119  input = input.Strip(TString::kBoth, ' ');
120  input = input.Strip(TString::kBoth, '\t');
121  fDrawOption = input;
123  }

207  const char* BranchPlotOptions::VarExp() const
208  {
210  if(!fTreeName.Length()) return nullptr;
211  if(!fListOfBranches.size()) return nullptr;
213  std::string varexp;
214  bool setbinning { true };

226  varexp += "(";

228  varexp += Form("%d,%g,%g,", branch.NBins(), branch.From(), branch.To());
