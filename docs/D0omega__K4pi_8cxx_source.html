239 {

244  RecMdcKalTrack::setPidType(RecMdcKalTrack::pion);

303 void D0omega_K4pi::WritePhotonKinematics(const AngleDifferences& angles)
304 {
305  if(!fNTuple_photon.DoWrite()) return;
306  double eraw = fTrackEMC->energy();
307  double phi = fTrackEMC->phi();
308  double theta = fTrackEMC->theta();
309 
310  HepLorentzVector four_mom(eraw * sin(theta) * cos(phi), eraw * sin(theta) * sin(phi),
311  eraw * cos(theta), eraw);
312  fNTuple_photon.GetItem<double>("E") = four_mom.e();
313  fNTuple_photon.GetItem<double>("px") = four_mom.px();
314  fNTuple_photon.GetItem<double>("py") = four_mom.py();
315  fNTuple_photon.GetItem<double>("pz") = four_mom.pz();
316  fNTuple_photon.GetItem<double>("smallest_phi") = angles.GetTheta();
317  fNTuple_photon.GetItem<double>("smallest_theta") = angles.GetPhi();
318  fNTuple_photon.GetItem<double>("smallest_angle") = angles.GetAngle();
319  fNTuple_photon.Write();
320 }
