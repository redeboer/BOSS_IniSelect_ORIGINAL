232 {
233  fTrackKal = track->mdcKalTrack();

237  RecMdcKalTrack::setPidType(RecMdcKalTrack::pion);

300 void D0omega_K4pi::GetPhotonAngles()
301 {
302  fAngle = fExtendedEmcPosition.angle(fEmcPosition);
303  fTheta = fExtendedEmcPosition.theta() - fEmcPosition.theta();
304  fPhi = fExtendedEmcPosition.deltaPhi(fEmcPosition);
305  fTheta = fmod(fTheta + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
306  fPhi = fmod(fPhi + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
307 }

327 void D0omega_K4pi::WritePhotonKinematics()
328 {
329  if(!fNTuple_photon.DoWrite()) return;
330  double eraw = fTrackEMC->energy();
331  double phi = fTrackEMC->phi();
332  double theta = fTrackEMC->theta();
333 
334  HepLorentzVector four_mom(eraw * sin(theta) * cos(phi), eraw * sin(theta) * sin(phi),
335  eraw * cos(theta), eraw);
336  fNTuple_photon.GetItem<double>("E") = four_mom.e();
337  fNTuple_photon.GetItem<double>("px") = four_mom.px();
338  fNTuple_photon.GetItem<double>("py") = four_mom.py();
339  fNTuple_photon.GetItem<double>("pz") = four_mom.pz();
340  fNTuple_photon.GetItem<double>("smallest_phi") = fSmallestTheta;
341  fNTuple_photon.GetItem<double>("smallest_theta") = fSmallestPhi;
342  fNTuple_photon.GetItem<double>("smallest_angle") = fSmallestAngle;
343  fNTuple_photon.Write();
344 }
