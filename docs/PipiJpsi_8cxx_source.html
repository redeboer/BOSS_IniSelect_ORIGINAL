541  // * Test event rate *
542  int m_trig_tot(0);

559  ++m_cout_col;
560  if(evtRecEvent->totalCharged() < 3 || evtRecTrkCol->size() < 3 ||
561  evtRecEvent->totalTracks() > 99 || evtRecTrkCol->size() > 99)
562  return StatusCode::SUCCESS;

568  Vint iGood;
569  // Number of different particles: \f$\pi^+\f$, \f$\pi^-\f$, \f$l^+\f$, \f$l^-\f$
570  int m_num[4] = {0, 0, 0, 0};
571  int nCharge = 0;
572  m_pion_matched = 0;
573  m_lep_matched = 0;
574  HepLorentzVector m_lv_pionp;

577  HepLorentzVector m_lv_lepm;

585  for(int i = 0; i < evtRecEvent->totalCharged(); ++i)
586  {

589  if(!(*itTrk)->isMdcKalTrackValid()) continue;
590  RecMdcKalTrack* mdcTrk = (*itTrk)->mdcKalTrack();
591 
593  m_vx0 = mdcTrk->x();
594  m_vy0 = mdcTrk->y();
595  m_vz0 = mdcTrk->z();
596  m_vr0 = mdcTrk->r();
597 
599  if(fabs(m_vz0) >= m_vz0cut) continue;
600  if(m_vr0 >= m_vr0cut) continue;
601 
603  iGood.push_back(i);
604  nCharge += mdcTrk->charge();
605 
608  if((*itTrk)->isEmcShowerValid())
609  {
610  if(mdcTrk->p() < 1.)
611  ++m_pion_matched;
612  else
613  ++m_lep_matched;
614  }
615 
617  if(mdcTrk->charge() > 0)
618  {
619  if(mdcTrk->p() < 1.0)
620  { // pi+
621  mdcTrk->setPidType(RecMdcKalTrack::pion);
622  // @warning If you do not modify the `DstMdcKalTrack` package, the `RecMdcKalTrack::p4()`
623  // function cannot be used. You should get the momentum from `MdcKalTrack`, then calculate
624  // the energy by yourself.
625  m_lv_pionp = mdcTrk->p4(xmass[2]);
626  ++m_num[0];
627  }
628  else
629  { // l+
630  mdcTrk->setPidType(RecMdcKalTrack::electron);
631  m_lv_pos = mdcTrk->p4(xmass[0]);
632  mdcTrk->setPidType(RecMdcKalTrack::muon);
633  m_lv_mup = mdcTrk->p4(xmass[1]);
634  ++m_num[2];
635  }
636  }
637  else
638  {
639  if(mdcTrk->p() < 1.0)
640  { // pi-
641  mdcTrk->setPidType(RecMdcKalTrack::pion);
642  m_lv_pionm = mdcTrk->p4(xmass[2]);
643  ++m_num[1];
644  }
645  else
646  { // l-
647  mdcTrk->setPidType(RecMdcKalTrack::electron);
648  m_lv_ele = mdcTrk->p4(xmass[0]);
649  mdcTrk->setPidType(RecMdcKalTrack::muon);
650  m_lv_mum = mdcTrk->p4(xmass[1]);
651  ++m_num[3];
652  }
653  }
654  }
656 
660  int nGood = iGood.size();
661  log << MSG::DEBUG << "With KalmanTrack, ngood, totcharge = " << nGood << " , " << nCharge
662  << endmsg;
663  if(nGood < 3 || nGood > 4) return sc;
664  ++m_cout_nGood;
665 
667  m_ep_ratio = 0;
668  for(int i = 0; i < evtRecEvent->totalTracks(); ++i)
669  {
670  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
671  if(!(*itTrk)->isEmcShowerValid()) continue;
672  RecEmcShower* emcTrk = (*itTrk)->emcShower();
673  m_ep_ratio += emcTrk->energy();
674  }
675 
677  if(m_ep_ratio > m_distin_emuon)
678  { // if electron
679  m_lv_lepp = m_lv_pos;
680  m_lv_lepm = m_lv_ele;
681  }
682  else
683  { // if muon
684  m_lv_lepp = m_lv_mup;
685  m_lv_lepm = m_lv_mum;
686  }
687 
689  HepLorentzVector m_lv_lab(0.04, 0, 0, 3.686);
690  if(nGood == 4)
691  {
692  if(nCharge) return sc;
693  m_event_flag = 4;
694  }
695  else
696  {
697  if(m_num[0] > 1 || m_num[1] > 1 || m_num[2] > 1 || m_num[3] > 1) return sc;
698  if(m_num[0] == 0)
