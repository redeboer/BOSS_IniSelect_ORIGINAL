223

228  fBckParameters.add(*p);
229  }
230  RooChebychev fPolBackground("polBkg",
231  Form("Polynomial-%u background", NPol()),
232  *fRooRealVar, fBckParameters);
233  RooRealVar fSigToBckRatio(
234  Form("N_{pol%u}", NPol()),
235  Form("N_{pol%u}", NPol()),
236  0., 0., 1e5);
237  if(NPol()) {
238  fComponents.add(fPolBackground);
239  fNContributions.add(fSigToBckRatio);
240  }
241 
242  // * Add the components and fit * //
243  RooAddPdf fFullShape("full_shape", "Double gaussian + background", fComponents, fNContributions);
244  auto result = fFullShape.fitTo(

246  RooFit::Range(FitFrom(), FitUntil()));

262 
263  // * Write fitted histograms * //
264  TString pname = Name();
