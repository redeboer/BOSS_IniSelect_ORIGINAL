177  Double_t sigmaErr{0.};
178  for(auto &rooVar : fitParameters[i]) {
179  auto value = rooVar->getVal();
180  TString name{rooVar->GetName()};
181  TString vstr{Form("%g", value)};
182  if(name.Length() > width_name) width_name = name.Length();
183  if(vstr.Length() > width_val ) width_val = vstr.Length();
184  if(name.Contains("sigma")) {
185  sigma += value*value;

187  }
188  }

193  << std::left << std::fixed << rooVar->getVal() << " \\pm "
