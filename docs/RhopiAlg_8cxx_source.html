482  MsgStream log(msgSvc(), name());
483  log << MSG::INFO << "In execute():" << endmsg;

487 
488  // * Load DST file info *
489  SmartDataPtr<Event::EventHeader> eventHeader(eventSvc(), "/Event/EventHeader");
490  int runNo = eventHeader->runNumber();
491  int evtNo = eventHeader->eventNumber();
492  log << MSG::DEBUG << "run, evtnum = " << runNo << " , " << evtNo << endmsg;
493  cout << "event number: " << evtNo << endl;
494  Ncut0++; // counter for all events
495 
496  // * Load event information and track collection *
497  /*
498  http://bes3.to.infn.it/Boss/7.0.2/html/namespaceEventModel_1_1EvtRec.html (namespace)
499  http://bes3.to.infn.it/Boss/7.0.2/html/classEvtRecEvent.html (class)
500  http://bes3.to.infn.it/Boss/7.0.2/html/EvtRecTrack_8h.html (typedef EvtRecTrackCol)
501  */
502  SmartDataPtr<EvtRecEvent> evtRecEvent(eventSvc(), EventModel::EvtRec::EvtRecEvent);
503  SmartDataPtr<EvtRecTrackCol> evtRecTrkCol(eventSvc(), EventModel::EvtRec::EvtRecTrackCol);
504 
505  // * Log number of events *
506  log << MSG::DEBUG << "Ncharged, Nneutral, Ntotal = " << evtRecEvent->totalCharged() << ", "
507  << evtRecEvent->totalNeutral() << ", " << evtRecEvent->totalTracks() << endmsg;
508 
510  Hep3Vector xorigin(0, 0, 0);
511  IVertexDbSvc* vtxsvc;
512  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
513  if(vtxsvc->isVertexValid())
514  {
515  double* dbv = vtxsvc->PrimaryVertex();
516  double* vv = vtxsvc->SigmaPrimaryVertex();
517  // HepVector dbv = fReader.PrimaryVertex(runNo);
518  // HepVector vv = fReader.SigmaPrimaryVertex(runNo);
519  xorigin.setX(dbv[0]);
520  xorigin.setY(dbv[1]);
521  xorigin.setZ(dbv[2]);
522  }
523 
526  // The first part of the set of reconstructed tracks are the charged tracks
527  int nCharge = 0; // Number of charged tracks as identified by the MDC.
528  Vint
529  iGood; // Vector of integers that give the position of tracks in the `evtRecEvent` marked good.
530  for(int i = 0; i < evtRecEvent->totalCharged(); ++i)
531  {
532 
533  // * Get track: beginning of all tracks + event number
534  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
535  if(!(*itTrk)->isMdcTrackValid()) continue;
536 
537  // * Get track info from Main Drift Chamber
538  RecMdcTrack* mdcTrk = (*itTrk)->mdcTrack();
539  double pch = mdcTrk->p();
540  double x0 = mdcTrk->x();
541  double y0 = mdcTrk->y();
542  double z0 = mdcTrk->z();
543  double phi0 = mdcTrk->helix(1);
544 
545  // * Get vertex origin
546  double xv = xorigin.x();
547  double yv = xorigin.y();
548  double Rxy = (x0 - xv) * cos(phi0) + (y0 - yv) * sin(phi0);
549 
550  // * Get radii of vertex
551  HepVector a = mdcTrk->helix();
552  HepSymMatrix Ea = mdcTrk->err();
553  HepPoint3D point0(0., 0., 0.); // the initial point for MDC recosntruction
554  HepPoint3D IP(xorigin[0], xorigin[1], xorigin[2]);
555  VFHelix helixip(point0, a, Ea);
556  helixip.pivot(IP);
557  HepVector vecipa = helixip.a();
558  double Rvxy0 = fabs(vecipa[0]); // nearest distance to IP in xy plane
559  double Rvz0 = vecipa[3]; // nearest distance to IP in z direction
560  double Rvphi0 = vecipa[1];
561 
562  // * WRITE primary vertex position info ("vxyz" branch) *
563  fVx0 = x0; // primary x-vertex as determined by MDC
564  fVy0 = y0; // primary y-vertex as determined by MDC
565  fVz0 = z0; // primary z-vertex as determined by MDC
566  fVr0 = Rxy; // distance from origin in xy-plane
567  fRvxy0 = Rvxy0; // nearest distance to IP in xy plane
568  fRvz0 = Rvz0; // nearest distance to IP in z direction
569  fRvphi0 = Rvphi0; // angle in the xy-plane (?)
570  fTupleVxyz->write(); // "vxyz" branch
571 
572  // * Apply vertex cuts *
573  if(fabs(z0) >= fVz0cut) continue;
574  if(fabs(Rxy) >= fVr0cut) continue;
575  if(fabs(Rvz0) >= fRvz0cut) continue;
576  if(fabs(Rvxy0) >= fRvxy0cut) continue;
577 
578  // * Add charged track to vector *
579  iGood.push_back(i);
580  nCharge += mdcTrk->charge();
581  }
582 
583  // * Finish Good Charged Track Selection * //
584  int nGood = iGood.size();
585  log << MSG::DEBUG << "ngood, totcharge = " << nGood << " , " << nCharge << endmsg;
586  if((nGood != 2) || (nCharge != 0)) { return StatusCode::SUCCESS; }
587  Ncut1++; // nGood!=2 or nCharge!=0
588 
592  Vint iGam;
593  for(int i = evtRecEvent->totalCharged(); i < evtRecEvent->totalTracks(); ++i)
594  {
595 
596  // * Get track
597  EvtRecTrackIterator itTrk = evtRecTrkCol->begin() + i;
598  if(!(*itTrk)->isEmcShowerValid()) continue;
599  RecEmcShower* emcTrk = (*itTrk)->emcShower();
600  Hep3Vector emcpos(emcTrk->x(), emcTrk->y(), emcTrk->z());
601 
602  // * Find the theta, phi, and angle difference with nearest charged track
603  double dthe = 200.; // start value for difference in theta
604  double dphi = 200.; // start value for difference in phi
605  double dang = 200.; // start value for difference in angle (?)
606  for(int j = 0; j < evtRecEvent->totalCharged(); j++)
607  {
608  EvtRecTrackIterator jtTrk = evtRecTrkCol->begin() + j;
609  if(!(*jtTrk)->isExtTrackValid()) continue;
610  RecExtTrack* extTrk = (*jtTrk)->extTrack();
611  if(extTrk->emcVolumeNumber() == -1) continue;
612  Hep3Vector extpos = extTrk->emcPosition();
613  // double ctht = extpos.cosTheta(emcpos);
614  double angd = extpos.angle(emcpos);
615  double thed = extpos.theta() - emcpos.theta();
616  double phid = extpos.deltaPhi(emcpos);
617  thed = fmod(thed + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
618  phid = fmod(phid + CLHEP::twopi + CLHEP::twopi + pi, CLHEP::twopi) - CLHEP::pi;
619  if(angd < dang)
620  {
621  dang = angd;
622  dthe = thed;
623  dphi = phid;
624  }
625  }
626 
627  // * Apply angle cut
628  if(dang >= 200) continue;
629  double eraw = emcTrk->energy();
630  dthe = dthe * 180 / (CLHEP::pi);
631  dphi = dphi * 180 / (CLHEP::pi);
632  dang = dang * 180 / (CLHEP::pi);
633 
634  // * WRITE photon info ("photon" branch)
635  fDeltaTheta = dthe; // theta difference with nearest charged track (degrees)
636  fDeltaPhi = dphi; // phi difference with nearest charged track (degrees)
637  fDeltaAngle = dang; // angle difference with nearest charged track
638  fEraw = eraw; // energy of the photon
639  fTupleAngles->write(); // "photon" branch
640 
641  // * Apply photon cuts
642  if(eraw < fEnergyThreshold) continue;
643  if((fabs(dthe) < fGammaThetaCut) && (fabs(dphi) < fGammaPhiCut)) continue;
644  if(fabs(dang) < fGammaAngleCut) continue;
645 
646  // * Add photon track to vector
647  iGam.push_back(i);
648  }
649 
650  // * Finish Good Photon Selection *
651  int nGam = iGam.size();
652  log << MSG::DEBUG << "Number of good photons: " << nGam << " , " << evtRecEvent->totalNeutral()
653  << endmsg;
654  if(nGam < 2) { return StatusCode::SUCCESS; }
655  Ncut2++; // number of photons < 2
656 
658  if(fCheckDedx)
659  {
660  for(int i = 0; i < nGood; ++i)
661  {
662
