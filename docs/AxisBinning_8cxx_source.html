103  void AxisBinning::Set(const std::string &binning)
104  {
106  if(!binning.size()) return;
107  int ncommas{0};
108  for(auto& c : binning) if(c==',') ++ncommas;
109  if(ncommas!=2) {
110  TerminalIO::PrintWarning(Form("Binning string \"%s\" has to contain 2 commas", binning.c_str()));
111  return;
112  }
114  int nbins = 0;
115  double width = 0.;
116  double from = 0.;
117  double to = 0.;
119  TString line(binning.c_str());
120  TString tok;
121  UChar_t sw = 0;
122  Ssiz_t start{0};
123  while(line.Tokenize(tok, start, ",")) {
124  switch(sw) {
125  case 0 :
126  if(tok.Contains('.') || tok.Contains('e') || tok.Contains('E')) width = tok.Atof();
127  else nbins = tok.Atoi();
128  break;
129  case 1 : from = tok.Atof(); break;
130  case 2 : to = tok.Atof(); break;
131  default :
132  TerminalIO::PrintFatalError("AxisBinning::Set(const std::string&)");
133  std::terminate();
134  }
135  ++sw;
136  }
138  if(nbins) Set(nbins, from, to); // if first argument is int
139  else Set(width, from, to); // if first argument is double
140  }
