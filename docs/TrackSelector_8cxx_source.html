251 StatusCode TrackSelector::execute()

264  fEventHeader = SmartDataPtr<Event::EventHeader>(eventSvc(), "/Event/EventHeader");
265  fEvtRecEvent = SmartDataPtr<EvtRecEvent>(eventSvc(), EventModel::EvtRec::EvtRecEvent);

270  fLog << MSG::DEBUG << "Ncharged = " << fEvtRecEvent->totalCharged() << ", "

280  Gaudi::svcLocator()->service("VertexDbSvc", vtxsvc);
281  if(vtxsvc->isVertexValid())

283  double* dbv = vtxsvc->PrimaryVertex();

285  fVertexPoint.set(dbv[0], dbv[1], dbv[2]);

288 
290  CreateChargedCollection();
291  CreateNeutralCollection();
292 
296  if(fNTuple_mult.DoWrite())

298  fNTuple_mult.GetItem<int>("Ntotal") = fEvtRecEvent->totalTracks();

456 void TrackSelector::CreateChargedCollection()
457 {
461  if(!fCreateChargedCollection) return;
462 
464  fChargedTracks.clear();
465  fPIDInstance = ParticleID::instance();
466 
468  if(!fEvtRecEvent->totalCharged())
469 
472  fNetChargeMDC = 0;
473  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
474  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i)
475  {
478 
479  // * Get track info from Main Drift Chamber
480  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
481  fTrackIter = fEvtRecTrkCol->begin() + i;
482  if(!(*fTrackIter)->isMdcTrackValid()) continue;
483  ++fCounter_Nmdcvalid;
484  fTrackMDC = (*fTrackIter)->mdcTrack();
485 
486  // * Get kinematics of track
487  double phi = fTrackMDC->helix(1);
488  double vr = (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
489  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
490 
491  // * Get radii of track vertex
492  HepVector a = fTrackMDC->helix();
493  HepSymMatrix Ea = fTrackMDC->err();
494  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
495  VFHelix helixip(point0, a, Ea);
496  helixip.pivot(fVertexPoint);
498  HepVector vecipa = helixip.a();
499  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
500  double rvz = vecipa[3]; // nearest distance to IP in z direction
501  double rvphi = vecipa[1]; // angle in the xy-plane (?)
502 
505  if(fCut_Vz.FailsMax(fabs(fTrackMDC->z())))
506  continue;
507  if(fCut_Vxy.FailsMax(vr))
508  continue;
509  if(fCut_Rz.FailsMax(fabs(rvz)))
510  continue;
511  if(fCut_Rxy.FailsMax(rvxy))
513  continue;
514  if(fCut_CosTheta.FailsMax(fabs(cos(fTrackMDC->theta()))))
516  continue;
517 
520  // * Add charged track to vector
521  fChargedTracks.push_back(*fTrackIter);
522  fNetChargeMDC += fTrackMDC->charge();
523 
525  if(fNTuple_charged.DoWrite())
526  {
527  fNTuple_charged.GetItem<double>("vx") = fTrackMDC->x();
528  fNTuple_charged.GetItem<double>("vy") = fTrackMDC->y();
529  fNTuple_charged.GetItem<double>("vz") = fTrackMDC->z();
530  fNTuple_charged.GetItem<double>("vr") = vr;
531  fNTuple_charged.GetItem<double>("rvxy") = rvxy;
532  fNTuple_charged.GetItem<double>("rvz") = rvz;
533  fNTuple_charged.GetItem<double>("rvphi") = rvphi;
534  fNTuple_charged.GetItem<double>("phi") = phi;
535  fNTuple_charged.GetItem<double>("p") = fTrackMDC->p();
536  fNTuple_charged.Write();
537  }
538 
540  WriteDedxInfo(*fTrackIter, fNTuple_dedx);
541 
543  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite())
544  {
545  // * Check if MDC and TOF info for track are valid *
546  if(!(*fTrackIter)->isMdcTrackValid()) continue;
547  if(!(*fTrackIter)->isTofTrackValid()) continue;
548  // * Get momentum as determined by MDC *
549  fTrackMDC = (*fTrackIter)->mdcTrack();
550  double ptrk;
551  if(fTrackMDC) ptrk = fTrackMDC->p();
552  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIter)->tofTrack();
553  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
554  for(; iter_tof != tofTrkCol.end(); ++iter_tof)
555  {
556  TofHitStatus hitStatus;
557  hitStatus.setStatus((*iter_tof)->status());
558  if(!hitStatus.is_counter()) continue;
559  if(hitStatus.is_barrel())
560  {
561  if(hitStatus.layer() == 1)
562  { // inner barrel
563  WriteTofInformation(iter_tof, ptrk, fNTuple_TofIB);
564  }
565  else if(hitStatus.layer() == 2)
566  { // outer barrel
567  WriteTofInformation(iter_tof, ptrk, fNTuple_TofOB);
568  }
569  }
570  else if(hitStatus.layer() == 0) // end cap
571  WriteTofInformation(iter_tof, ptrk, fNTuple_TofEC);
572  }
573  }
574 
576  }
577 
578  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fChargedTracks.size() << endmsg;
580 }

861 void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk,
862  NTupleContainer& tuple)
863 {
865  if(!tuple.DoWrite()) return;
866 
868  double path = (*iter_tof)->path();
869  std::vector<double> texp(gNMasses);
870  for(size_t j = 0; j < texp.size(); ++j)
871  {
872  double gb = ptrk / gMasses[j]; // v = p/m (non-relativistic velocity)
873  double beta = gb / sqrt(1 + gb * gb);
874  texp[j] = 10 * path / beta / gSpeedOfLight; // hypothesis ToF
875  }
876 
878  tuple.GetItem<double>("p") = ptrk;
879  tuple.GetItem<double>("tof") = (*iter_tof)->tof();
880  tuple.GetItem<double>("path") = path;
881  tuple.GetItem<double>("cntr") = (*iter_tof)->tofID();
882  tuple.GetItem<double>("ph") = (*iter_tof)->ph();
883  tuple.GetItem<double>("zrhit") = (*iter_tof)->zrhit();
884  tuple.GetItem<double>("qual") = (*iter_tof)->quality();
885  tuple.GetItem<double>("tof_e") = path - texp[0];
886  tuple.GetItem<double>("tof_mu") = path - texp[1];
887  tuple.GetItem<double>("tof_pi") = path - texp[2];
888  tuple.GetItem<double>("tof_K") = path - texp[3];
889  tuple.GetItem<double>("tof_p") = path - texp[4];
890 
892  tuple.Write();
893 }
