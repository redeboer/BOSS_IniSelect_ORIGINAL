459 void TrackSelector::CreateChargedCollection()
460 {
461  if(!fCreateChargedCollection) return;
462  fChargedTracks.clear();
463  if(!fEvtRecEvent->totalCharged()) return;
464  fPIDInstance = ParticleID::instance();
466  fNetChargeMDC = 0;
467  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
468  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i)
469  {
470  SetTrackIter(i);
471  if(!IsMdcTrackValid()) continue;
472  fTrackMDC = (*fTrackIter)->mdcTrack();
473  fSecondaryVtx.SetValues(fTrackMDC, fVertexPoint);
474  if(!CutSecondaryVertex()) continue;
475 
476  // * Add charged track to vector
477  fChargedTracks.push_back(*fTrackIter);
478  fNetChargeMDC += fTrackMDC->charge();
479  WriteChargedTrackVertex();
480 
482  WriteDedxInfo(*fTrackIter, fNTuple_dedx);
483 
485  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite())
486  {
487  // * Check if MDC and TOF info for track are valid *
488  if(!(*fTrackIter)->isMdcTrackValid()) continue;
489  if(!(*fTrackIter)->isTofTrackValid()) continue;
490  // * Get momentum as determined by MDC *
491  fTrackMDC = (*fTrackIter)->mdcTrack();
492  double ptrk;
493  if(fTrackMDC) ptrk = fTrackMDC->p();
494  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIter)->tofTrack();
495  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
496  for(; iter_tof != tofTrkCol.end(); ++iter_tof)
497  {
498  TofHitStatus hitStatus;
499  hitStatus.setStatus((*iter_tof)->status());
500  if(!hitStatus.is_counter()) continue;
501  if(hitStatus.is_barrel())
502  {
503  if(hitStatus.layer() == 1) // inner barrel
504  WriteTofInformation(iter_tof->data(), ptrk, fNTuple_TofIB);
505  else if(hitStatus.layer() == 2) // outer barrel
506  WriteTofInformation(iter_tof->data(), ptrk, fNTuple_TofOB);
507  }
508  else if(hitStatus.layer() == 0) // end cap
509  WriteTofInformation(iter_tof->data(), ptrk, fNTuple_TofEC);
510  }
511  }
512  }
513 
514  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fChargedTracks.size() << endmsg;
515 }

591 void TrackSelector::WriteTofInformation(RecTofTrack* tofTrack, double ptrk, NTupleContainer& tuple)
592 {
594  if(!tuple.DoWrite()) return;
595 
597  double path = tofTrack->path();
598  std::vector<double> texp(Mass::TOF.size());
599  for(size_t j = 0; j < texp.size(); ++j)
600  {
601  double gb = ptrk / Mass::TOF.at(j); // v = p/m (non-relativistic velocity)
602  double beta = gb / sqrt(1 + gb * gb);
603  texp[j] = 10 * path / beta / gSpeedOfLight; // hypothesis ToF
604  }
605 
607  tuple.GetItem<double>("p") = ptrk;
608  tuple.GetItem<double>("tof") = tofTrack->tof();
609  tuple.GetItem<double>("path") = path;
610  tuple.GetItem<double>("cntr") = tofTrack->tofID();
611  tuple.GetItem<double>("ph") = tofTrack->ph();
612  tuple.GetItem<double>("zrhit") = tofTrack->zrhit();
613  tuple.GetItem<double>("qual") = tofTrack->quality();
614  tuple.GetItem<double>("tof_e") = path - texp[0];
615  tuple.GetItem<double>("tof_mu") = path - texp[1];
616  tuple.GetItem<double>("tof_pi") = path - texp[2];
617  tuple.GetItem<double>("tof_K") = path - texp[3];
618  tuple.GetItem<double>("tof_p") = path - texp[4];
619 
621  tuple.Write();
622 }
