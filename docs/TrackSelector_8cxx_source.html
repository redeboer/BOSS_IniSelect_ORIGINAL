461 void TrackSelector::CreateChargedCollection()
462 {
466  if(!fCreateChargedCollection) return;
467 
469  fGoodChargedTracks.clear();
470  fPIDInstance = ParticleID::instance();
471 
473  if(!fEvtRecEvent->totalCharged())
474 
477  fNetChargeMDC = 0;
478  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
479  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i)
480  {
483 
484  // * Get track info from Main Drift Chamber
485  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
486  fTrackIterator = fEvtRecTrkCol->begin() + i;
487  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
488  ++fCounter_Nmdcvalid;
489  fTrackMDC = (*fTrackIterator)->mdcTrack();
490 
491  // * Get kinematics of track
492  double phi = fTrackMDC->helix(1);
493  double vr = (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
494  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
495 
496  // * Get radii of track vertex
497  HepVector a = fTrackMDC->helix();
498  HepSymMatrix Ea = fTrackMDC->err();
499  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
500  VFHelix helixip(point0, a, Ea);
501  helixip.pivot(
502  fVertexPoint);
503  HepVector vecipa = helixip.a();
504  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
505  double rvz = vecipa[3]; // nearest distance to IP in z direction
506  double rvphi = vecipa[1]; // angle in the xy-plane (?)
507 
510  if(fCut_Vz.FailsMax(fabs(fTrackMDC->z())))
511  continue;
512  if(fCut_Vxy.FailsMax(vr))
513  continue;
514  if(fCut_Rz.FailsMax(fabs(rvz)))
515  continue;
516  if(fCut_Rxy.FailsMax(rvxy))
518  continue;
519  if(fCut_CosTheta.FailsMax(fabs(cos(fTrackMDC->theta()))))
521  continue;
522 
525  // * Add charged track to vector
526  fGoodChargedTracks.push_back(*fTrackIterator);
527  fNetChargeMDC += fTrackMDC->charge();
528 
530  if(fNTuple_charged.DoWrite())
531  {
532  fNTuple_charged.GetItem<double>("vx") = fTrackMDC->x();
533  fNTuple_charged.GetItem<double>("vy") = fTrackMDC->y();
534  fNTuple_charged.GetItem<double>("vz") = fTrackMDC->z();
535  fNTuple_charged.GetItem<double>("vr") = vr;
536  fNTuple_charged.GetItem<double>("rvxy") = rvxy;
537  fNTuple_charged.GetItem<double>("rvz") = rvz;
538  fNTuple_charged.GetItem<double>("rvphi") = rvphi;
539  fNTuple_charged.GetItem<double>("phi") = phi;
540  fNTuple_charged.GetItem<double>("p") = fTrackMDC->p();
541  fNTuple_charged.Write();
542  }
543 
545  WriteDedxInfo(*fTrackIterator, fNTuple_dedx);
546 
548  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite())
549  {
550  // * Check if MDC and TOF info for track are valid *
551  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
552  if(!(*fTrackIterator)->isTofTrackValid()) continue;
553  // * Get momentum as determined by MDC *
554  fTrackMDC = (*fTrackIterator)->mdcTrack();
555  double ptrk;
556  if(fTrackMDC) ptrk = fTrackMDC->p();
557  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
558  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
559  for(; iter_tof != tofTrkCol.end(); ++iter_tof)
560  {
561  TofHitStatus hitStatus;
562  hitStatus.setStatus((*iter_tof)->status());
563  if(!hitStatus.is_counter()) continue;
564  if(hitStatus.is_barrel())
565  {
566  if(hitStatus.layer() == 1)
567  { // inner barrel
568  WriteTofInformation(iter_tof, ptrk, fNTuple_TofIB);
569  }
570  else if(hitStatus.layer() == 2)
571  { // outer barrel
572  WriteTofInformation(iter_tof, ptrk, fNTuple_TofOB);
573  }
574  }
575  else if(hitStatus.layer() == 0) // end cap
576  WriteTofInformation(iter_tof, ptrk, fNTuple_TofEC);
577  }
578  }
579 
581  }
582 
583  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fGoodChargedTracks.size() << endmsg;
585 }

886 void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk,
887  NTupleContainer& tuple)
888 {
890  if(!tuple.DoWrite()) return;
891 
893  double path = (*iter_tof)->path();
894  std::vector<double> texp(gNMasses);
895  for(size_t j = 0; j < texp.size(); ++j)
896  {
897  double gb = ptrk / gMasses[j]; // v = p/m (non-relativistic velocity)
898  double beta = gb / sqrt(1 + gb * gb);
899  texp[j] = 10 * path / beta / gSpeedOfLight; // hypothesis ToF
900  }
901 
903  tuple.GetItem<double>("p") = ptrk;
904  tuple.GetItem<double>("tof") = (*iter_tof)->tof();
905  tuple.GetItem<double>("path") = path;
906  tuple.GetItem<double>("cntr") = (*iter_tof)->tofID();
907  tuple.GetItem<double>("ph") = (*iter_tof)->ph();
908  tuple.GetItem<double>("zrhit") = (*iter_tof)->zrhit();
909  tuple.GetItem<double>("qual") = (*iter_tof)->quality();
910  tuple.GetItem<double>("tof_e") = path - texp[0];
911  tuple.GetItem<double>("tof_mu") = path - texp[1];
912  tuple.GetItem<double>("tof_pi") = path - texp[2];
913  tuple.GetItem<double>("tof_K") = path - texp[3];
914  tuple.GetItem<double>("tof_p") = path - texp[4];
915 
917  tuple.Write();
