226  const bool CommonFunctions::String::IsComment(std::string input)
227  {
228  Trim(input);
229  input = input.substr(0,2);
230  if(!input.compare("//")) return true; // C++ line comment
231  if(!input.compare("/*")) return true; // C++ begin block comment
232  if(input.front() == '*') return true; // C++ continue/end block comment
233  if(input.front() == '#') return true; // bash comment
234  if(input.front() == '%') return true; // LaTeX comment
235  return false;
236  }
