191  void CommonFunctions::Fit::FitBWGaussianConvolution(TH1F *hist, const ReconstructedParticle& particle, TString logScale)
192  {
193 
194  // * Create RooFit variable and data distribution * //
195  RooRealVar fRooRealVar = CreateRooFitInvMassVar(particle);
196  RooDataHist fRooDataHist = CreateRooFitInvMassDistr(hist, fRooRealVar, particle);
197 
198  // * Create double Gaussian function * //
199  RooRealVar m0("GaussianMeanZero", "GaussianMeanZero", 0.);
200  RooRealVar fSigma("#sigma_{gauss}", Form("%s Gaussian width", particle.NameLaTeX()),
201  particle.SingleGaussianWidth(),
202  Settings::Fit::gSigmaScaleFactorLow * particle.SingleGaussianWidth(),
203  Settings::Fit::gSigmaScaleFactorUp * particle.SingleGaussianWidth());
204  RooGaussian fGaussian("gauss",
205  Form("Gaussian PDF for #it{M}_{%s} distribution", particle.DaughterLabel()),
206  fRooRealVar, m0, fSigma);
207 
208  // * Create Breit-Wigner/Cauchy distribution * //
209  RooRealVar mean(
210  Form("m_{%s}", particle.NameLaTeX()),
211  Form("%s mass", particle.NameLaTeX()),
212  particle.Mass(), particle.LowerMass(), particle.UpperMass());
213  RooRealVar width("#sigma_{BW}", Form("%s BW width", particle.NameLaTeX()),
214  particle.BWConvolutedWidth(), 0., 10.*particle.BWConvolutedWidth());
215  RooBreitWigner bw("breitwigner",
216  Form("Breit-Wigner PDF for #it{M}_{%s} distribution", particle.DaughterLabel()),
217  fRooRealVar, mean, width);
218 
219  // * Convolute * //
220  RooFFTConvPdf signal("convolution", "convolution", fRooRealVar, bw, fGaussian);
221  RooRealVar n("N_{gaus1}", "N_{gaus1}", 1e2, 0., 1e6);
222  RooArgList fComponents(signal);
223  RooArgList fNContributions(n);
224 
225  // * Add polynomial background if required * //
226  RooArgList fBckParameters;
227  for(UChar_t i = 0; i <= particle.NPol(); ++i) {
228  auto p = new RooRealVar(Form("p%u", i), Form("p%u", i), 0., -1e6, 1e6);
229  fBckParameters.add(*p);
230  }
231  RooChebychev fPolBackground("polBkg",
232  Form("Polynomial-%u background", particle.NPol()),
233  fRooRealVar, fBckParameters);
234  RooRealVar fSigToBckRatio(
235  Form("N_{pol%u}", particle.NPol()),
236  Form("N_{pol%u}", particle.NPol()),
237  0., 0., 1e5);
238  if(particle.NPol()) {
239  fComponents.add(fPolBackground);
240  fNContributions.add(fSigToBckRatio);
241  }
242 
243  // * Add the components and fit * //
244  RooAddPdf fFullShape("full_shape", "Double gaussian + background", fComponents, fNContributions);
245  fFullShape.fitTo(
246  fRooDataHist,
247  RooFit::Range(particle.FitFrom(), particle.FitUntil()));
248 
249  // * Plot results and save * //
250  RooPlot *frame = fRooRealVar.frame(); // create a frame to draw
251  frame->SetAxisRange(particle.PlotFrom(), particle.PlotUntil());
252  fRooDataHist.plotOn(frame, // draw distribution
253  RooFit::LineWidth(2), RooFit::LineColor(kBlue+2), RooFit::LineWidth(1),
254  RooFit::MarkerColor(kBlue+2), RooFit::MarkerSize(.5));
255  fFullShape.plotOn(frame, RooFit::LineWidth(2), RooFit::LineColor(kBlack));
256  if(particle.NPol()) {
257  fFullShape.plotOn(frame, RooFit::Components(signal), // draw signal
258  RooFit::LineWidth(1), RooFit::LineColor(kRed-4));
259  fFullShape.plotOn(frame, RooFit::Components(fPolBackground), // draw background
260  RooFit::LineStyle(kDashed), RooFit::LineWidth(1), RooFit::LineColor(kGray));
261  }
262  fFullShape.paramOn(frame, RooFit::Layout(.56, .98, .92));
263 
264  // * Write fitted histograms * //
265  TString pname = particle.Name();
266  pname.ReplaceAll("/", ""); // in case of for instance "J/psi"
267  CommonFunctions::Draw::DrawAndSave(Form("ConvBWSingleGauss_%s", pname.Data()), "", logScale, frame);
268  delete frame;
269 
270  }
