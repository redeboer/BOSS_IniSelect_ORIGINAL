185 void TrackSelector::WriteTofInformation(RecTofTrack* tofTrack, double ptrk, NTupleContainer& tuple)
186 {
188  if(!tuple.DoWrite()) return;
189 
191  double path = tofTrack->path();
192  vector<double> texp(Mass::TOF.size());
193  for(size_t j = 0; j < texp.size(); ++j)
194  {
195  double gb = ptrk / Mass::TOF.at(j); // v = p/m (non-relativistic velocity)
196  double beta = gb / sqrt(1 + gb * gb);
197  texp[j] = 10 * path / beta / gSpeedOfLight; // hypothesis ToF
198  }
199 
201  tuple.GetItem<double>("p") = ptrk;
202  tuple.GetItem<double>("tof") = tofTrack->tof();
203  tuple.GetItem<double>("path") = path;
204  tuple.GetItem<double>("cntr") = tofTrack->tofID();
205  tuple.GetItem<double>("ph") = tofTrack->ph();
206  tuple.GetItem<double>("zrhit") = tofTrack->zrhit();
207  tuple.GetItem<double>("qual") = tofTrack->quality();
208  tuple.GetItem<double>("tof_e") = path - texp[0];
209  tuple.GetItem<double>("tof_mu") = path - texp[1];
210  tuple.GetItem<double>("tof_pi") = path - texp[2];
211  tuple.GetItem<double>("tof_K") = path - texp[3];
212  tuple.GetItem<double>("tof_p") = path - texp[4];
213 
215  tuple.Write();
216 }
